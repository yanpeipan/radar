         ```sql
           created_at TEXT DEFAULT CURRENT_TIMESTAMP
         )
         ```
       - Creates articles table:
         ```sql

           id TEXT PRIMARY KEY,
           feed_id TEXT NOT NULL REFERENCES feeds(id) ON DELETE CASCADE,

           link TEXT,
           guid TEXT NOT NULL,
           pub_date TEXT,
           description TEXT,
           content TEXT,
           created_at TEXT DEFAULT CURRENT_TIMESTAMP,
           UNIQUE(feed_id, guid)
         )
         ```
       - Creates indexes: idx_articles_feed_id, idx_articles_pub_date, idx_articles_link
