```python
def _load_feed_metadata_for_url(url: str) -> dict:
    """Load feed metadata for a URL to get path_filters."""
    from src.storage import get_feed as storage_get_feed, list_feeds as storage_list_feeds
    feeds = storage_list_feeds()
    for feed in feeds:
        if feed.url == url:
            if feed.metadata:
                import json
                try:
                    return json.loads(feed.metadata)
                except json.JSONDecodeError:
                    return {}
            break
    return {}
```
